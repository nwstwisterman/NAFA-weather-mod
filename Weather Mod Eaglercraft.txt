<script>
/* ===== CORE SYSTEMS ===== */
const UI={open:()=>menu.style.display="block",
close:()=>menu.style.display="none"};
const menu=document.getElementById("stormMenu");
setTimeout(()=>UI.open(),900);
document.addEventListener("keydown",e=>{
 if(e.key==="m")UI.open();
});

/* ===== SEASONS ===== */
const Seasons={cur:"summer",
isCold(){return this.cur==="winter"},
tornado(){return this.cur!=="winter"}};

/* ===== STORMS ===== */
const Storm={
settings:{intensity:1},
storms:[
{x:3000,z:-1000,i:.6,r:0,dx:.04,dz:.02,type:"supercell"},
{x:-4500,z:2000,i:.3,r:1.4,dx:.02,dz:-.03,type:"winter"}
],
update(){
 this.storms.forEach(s=>{
  s.x+=s.dx; s.z+=s.dz;
  s.r+=.002; s.i=Math.min(1,
   s.i+.0001*this.settings.intensity);
 });
},
closest(){
 return this.storms[0];
}
};

/* ===== WARNING HISTORY ===== */
const History={
log:[],
add(type,dur){
 this.log.unshift({type,start:Date.now(),dur});
 if(this.log.length>20)this.log.pop();
},
show(){
 alert(
 this.log.map(w=>
   `${w.type} (${Math.round((Date.now()-w.start)/60000)} min ago)`
 ).join("\n")||"No warnings yet"
 );
}
};

/* ===== AUDIO ===== */
const audio=new (window.AudioContext||webkitAudioContext)();
function EASSpeak(msg){
 const u=new SpeechSynthesisUtterance(msg);
 u.pitch=.1;u.rate=.85; speechSynthesis.speak(u);
}
let siren=null;
function sirenOn(){
 if(siren)return;
 siren=audio.createOscillator();
 const g=audio.createGain();
 siren.frequency.value=440;
 siren.connect(g);g.connect(audio.destination);
 g.gain.value=.12;siren.start();
}
function sirenOff(){if(siren){siren.stop();siren=null}}

/* ===== WARNINGS ===== */
const Alert={
cur:null,exp:null,
set(type,col,dur){
 if(this.cur===type)return;
 this.cur=type;this.exp=Date.now()+dur*60000;
 alertBox(type,col);History.add(type,dur);
 EASSpeak(`The National Weather Service has issued a ${type}`);
 if(type.includes("Tornado"))sirenOn();
},
clear(){
 this.cur=null;this.exp=null;sirenOff();hideAlert();
},
tick(){
 if(this.exp&&Date.now()>this.exp)this.clear();
}
};

const alertDiv=document.getElementById("stormAlert");
function alertBox(t,c){
 alertDiv.textContent=t;
 alertDiv.style.background=c;
 alertDiv.style.top="10px";
}
function hideAlert(){alertDiv.style.top="-60px"}

/* ===== RADAR ===== */
const Radar={
ctx:document.getElementById("radar").getContext("2d"),
show(){radar.style.display="block"}
};

/* ===== MAIN LOOP ===== */
function tick(){
 Storm.update();
 const s=Storm.closest();
 if(s.i>.85&&Seasons.tornado())
   Alert.set("🚨 Tornado Warning","red",10);
 else if(s.i>.6)
   Alert.set("Severe Thunderstorm Warning","orange",15);
 else Alert.clear();

 Alert.tick();
 requestAnimationFrame(tick);
}
tick();
</script>
